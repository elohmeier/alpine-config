alpine-config

Fleet configuration repository for Alpine Linux hosts.

Enrollment

# Add repositories and keys
echo "https://elohmeier.github.io/alpine-config/$(uname -m)" >> /etc/apk/repositories
echo "https://elohmeier.github.io/alpine-packages/$(uname -m)" >> /etc/apk/repositories
wget -qO /etc/apk/keys/melange.rsa.pub https://elohmeier.github.io/alpine-config/keys/melange.rsa.pub
wget -qO /etc/apk/keys/packages@elohmeier.rsa.pub https://elohmeier.github.io/alpine-packages/keys/packages@elohmeier.rsa.pub

# Install and enroll
apk update && apk add alpine-enroll
alpine-enroll
apk del alpine-enroll  # optional cleanup

apk update && apk add host-config
